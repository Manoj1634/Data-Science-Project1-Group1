# Function to convert volume strings to numeric
convert_volume <- function(vol) {
  multiplier <- 1
  if (grepl("M", vol)) {
    multiplier <- 1e6
    vol <- gsub("M", "", vol)
  } else if (grepl("B", vol)) {
    multiplier <- 1e9
    vol <- gsub("B", "", vol)
  } else if (grepl("K", vol)) {
    multiplier <- 1e3
    vol <- gsub("K", "", vol)
  }
  as.numeric(gsub(",", "", vol)) * multiplier
}
